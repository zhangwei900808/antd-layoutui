    |-- pages          # 页面组件目录
    |-- redux          # redux目录
    |   |-- actions    # actions 目录
    |   |-- middleware # 中间件目录
    |   |-- reducers   # reducers目录
    |   |-- sagas      # sagas 目录
    |   |-- store      # store 目录
    |-- router         # 路由 目录
    |   |-- index.js
    |-- service        # 后台 service 服务
    |   |-- apis       # apis 目录
    |   |   |-- 1.0
    |   |       |-- index.js
    |   |       |-- urls.js
    |   |-- mocks      # mocks 目录

        |-- App.test.js
